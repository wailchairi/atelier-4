#include <iostream>
using namespace std;

class Test {
private:
    static int compteur;

public:
    void call() {
        compteur++;
    }

    static int getCompteur() {
        return compteur;
    }
};
int Test::compteur = 0;

int main() {
    Test t1, t2, t3;

    t1.call();
    t2.call();
    t1.call();
    t3.call();
    t2.call();

    cout << "La fonction call a été appelée " << Test::getCompteur() << " fois." << endl;

    return 0;
}
